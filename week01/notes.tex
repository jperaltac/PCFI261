\begin{pythoncode}
from math import *
import matplotlib.pyplot as plt
from random import random

N, R = 100000, 3.0

def signrandom(): return 2.0*random()-1.0

X = [signrandom()*10.0 for i in range(N)]
Y = [signrandom()*10.0 for i in range(N)]

Xs, Ys = list(), list()
for i in range(N):
    if sqrt(X[i]*X[i]+Y[i]*Y[i]) < R:
       Xs.append(X[i])
       Ys.append(Y[i])

plt.scatter(Xs, Ys, s=1)
plt.xlabel("$x$")
plt.ylabel("$y$")
plt.show()
\end{pythoncode}

\begin{frame}[fragile]

\frametitle{Dinámica de crecimiento de poblaciones}

Primero supongamos que el número de individuos crece a una tasa proporcional al número actual,

\begin{equation}
\frac{dN(t)}{dt} = \lambda N(t).
\end{equation}

La solución es conocida y no de mucho interés,

\begin{equation}
N(t) = N(0)\exp(\lambda t),
\end{equation}
%
ya que para $\lambda > 0$ produce un crecimiento indefinido. Por supuesto, esto no sucede en 
la realidad, por lo que para mejorar el modelo haremos que la tasa de crecimiento se sature 
al llegar a una población máxima $N^*$.

\begin{equation}
\frac{dN(t)}{dt} = \lambda'(N^*-N(t))N(t).
\end{equation}

Con este cambio ya tenemos una ecuación no lineal, que manifiesta comportamientos interesantes.

\end{frame}

%
%
%
\begin{frame}[fragile]

\frametitle{Dinámica de crecimiento de poblaciones}
\framesubtitle{Versión discreta}

Dividiendo el tiempo en pasos finitos (``discretizando''), tenemos que

\begin{equation}
\frac{dN(t)}{dt} \approx \frac{N(t+\Delta t)-N(t)}{\Delta t},
\end{equation}
%
luego

\begin{equation}
\frac{N(t+\Delta t)-N(t)}{\Delta t} \approx \lambda'(N^* - N(t))N(t).
\end{equation}

Llamando $N(t)=N_i$ y $N(t+\Delta t)=N_{i+1}$, podemos escribirlo como una regla de evolución discreta,

\begin{equation}
N_{i+1} = N_i + \Delta t \lambda'(N^* - N_i)N_i.
\end{equation}

\end{frame}

%
%
%
\begin{frame}[fragile]

Factorizando $N_i(1+\lambda'\Delta tN^*)$, tenemos

\begin{equation}
N_{i+1} = N_i(1+\lambda'\Delta tN^*)\left[1-\frac{\lambda'\Delta t}{1+\lambda'\Delta tN^*}N_i\right].
\end{equation}

Notemos que, si $\lambda'\Delta t N^* \gg 1$, entonces 

\begin{equation}
\frac{\lambda'\Delta t}{1+\lambda'\Delta tN^*} \approx \frac{1}{N^*}.
\end{equation} 

Por comodidad definimos nuevas cantidades,

\begin{eqnarray}
\mu = 1+\lambda'\Delta tN^*, \\
x_i = \frac{N_i}{N^*},
\end{eqnarray}
%
para que nuestra regla de evolución quede directamente como

\begin{block}{Mapa logístico}
\begin{equation}
x_{i+1} = \mu x_i(1-x_i).
\end{equation}
\end{block}

\end{frame}

%
%
%
\begin{frame}[fragile]
\frametitle{Puntos fijos}

Un mapa $x \rightarrow f(x)$ puede bajo ciertas condiciones presentar \textbf{puntos fijos}:

\begin{block}{Punto fijo}
Valor $x_0$ tal que $f(x_0)=x_0$. Este punto evoluciona sobre sí mismo \\ bajo el mapa $f$.
\end{block}

En el caso del mapa logístico, los puntos fijos son por supuesto las soluciones de

\begin{equation}
x_0 = \mu x_0 (1 - x_0)
\end{equation}
%
que son $x_0=0$ (solución trivial) y $x_0=(\mu-1)/\mu$. Esto sin embargo es sólo una solución estable si

\begin{equation}
\left|\frac{df}{dx}\right|_{x_0}=\mu-2\mu x_0 < 1,
\end{equation}

En el caso en que $x_0=(\mu-1)/\mu$, la condición implica $\mu < 3$. \\

¿Qué sucede entonces si usamos $\mu \geq 3$? ?`Existe convergencia?

\end{frame}

%
%
%
\begin{frame}[fragile]
\frametitle{Diversos comportamientos de un mapa no lineal}

\begin{block}{Atractores}
Es posible que muchas semillas iniciales lleguen al mismo punto fijo. En este caso diremos
que el punto fijo es un \textbf{atractor}.
\end{block}

Para un conjunto de parámetros del mapa, es posible 

\begin{itemize}
\item Tener un único atractor 
\item Tener dos, tres o un número finito de atractores 
\item Tener \textbf{caos}, en ese caso el resultado depende tan finamente de la condición inicial que 
es imposible de preveer (a menos que calculemos la secuencia completa).
\end{itemize}

\begin{block}{}
Como veremos, el mapa logístico presenta todos estos comportamientos, dependiendo del valor de $\mu$.
\end{block}

\end{frame}

\begin{frame}[fragile]
\frametitle{Doblamiento de período y diagramas de bifurcación}

El mapa logístico llega a su región caótica por medio de un mecanismo conocido como doblamiento 
de período. El número de atractores se duplica hasta que éstos forman un continuo en la región caótica.
